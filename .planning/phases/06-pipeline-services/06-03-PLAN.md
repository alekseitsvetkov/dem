<objective>
Implement the Demo Downloader Service: a NATS JetStream pull consumer that receives download jobs, streams `.dem.gz` files from the HLTV CDN directly into MinIO (zero local disk writes), retries transient failures with exponential backoff, and publishes parse jobs to `dem.parse.jobs` on success.

Purpose: This is the middle stage of the data pipeline. The downloader bridges HLTV's CDN (unreliable, rate-limited) to MinIO (reliable, under our control). Per D-03, it retries locally before escalating to NATS redelivery. Per D-04, it never writes to local disk.

Output: `cmd/downloader/main.go`, `internal/downloader/config.go`, `internal/downloader/downloader.go`.
</objective>

   ```go
   func (d *DownloaderService) processMessage(ctx context.Context, msg jetstream.Msg) {
       defer msg.Ack()  // NON-NEGOTIABLE per D-10

       // Parse JSON payload from poller:
       // {match_id, demo_url, match_url, event_name, team1, team2, match_date, discovered_at}
       var job struct {
           MatchID   string `json:"match_id"`
           DemoURL   string `json:"demo_url"`
           MatchURL  string `json:"match_url"`
           EventName string `json:"event_name"`
           Team1     string `json:"team1"`
           Team2     string `json:"team2"`
           MatchDate string `json:"match_date"`
       }
       json.Unmarshal(msg.Data(), &job)

       logger := d.logger.With(slog.String("match_id", job.MatchID), slog.String("job_id", /* msg metadata ID */))

       // Download with retry (D-03: 3 attempts, 5s/25s/125s backoff)
       objectKey := fmt.Sprintf("demos/%s.dem.gz", job.MatchID)
       err := d.downloadWithRetry(ctx, job.DemoURL, d.cfg.MinioBucket, objectKey, logger)
       if err != nil {
           logger.Error("download failed after retries", slog.String("error", err.Error()))
           msg.NakWithDelay(d.cfg.NakDelay)  // escalate to NATS redelivery
           return
       }

       // Publish parse job (D-05)
       parseJob := map[string]string{
           "bucket":     d.cfg.MinioBucket,
           "object_key": objectKey,
           "match_id":   job.MatchID,
           "match_url":  job.MatchURL,
           "event_name": job.EventName,
           "team1":      job.Team1,
           "team2":      job.Team2,
           "match_date": job.MatchDate,
       }
       parseJSON, _ := json.Marshal(parseJob)
       d.js.PublishMsg(ctx, jetstream.PublishMsg{
           Subject: natsutil.SubjectParse,
           Data:    parseJSON,
       })
       logger.Info("published parse job")
   }
   ```

5. **downloadWithRetry(ctx, demoURL, bucket, objectKey, logger) — D-03, D-04:**
   ```go
   func (d *DownloaderService) downloadWithRetry(ctx context.Context, demoURL, bucket, objectKey string, logger *slog.Logger) error {
       var lastErr error
       for attempt := 0; attempt < d.cfg.MaxRetries; attempt++ {
           if attempt > 0 {
               delay := d.cfg.RetryBaseDelay * time.Duration(1<<uint(attempt-1))  // 5s, 25s, 125s
               if delay > d.cfg.RetryMaxDelay {
                   delay = d.cfg.RetryMaxDelay
               }
               // Add jitter: +/- 20% random
               jitter := time.Duration(rand.Int63n(int64(delay) / 5))
               time.Sleep(delay + jitter)
               logger.Info("retrying download", slog.Int("attempt", attempt+1), slog.Duration("delay", delay+jitter))
           }

           err := d.streamDownload(ctx, demoURL, bucket, objectKey)
           if err == nil {
               return nil
           }
           lastErr = err
           logger.Warn("download attempt failed", slog.Int("attempt", attempt+1), slog.String("error", err.Error()))
       }
       return fmt.Errorf("download failed after %d attempts: %w", d.cfg.MaxRetries, lastErr)
   }
   ```

6. **streamDownload — D-04 (zero local disk writes, streaming from HLTV CDN to MinIO):**
   ```go
   func (d *DownloaderService) streamDownload(ctx context.Context, demoURL, bucket, objectKey string) error {
       downloadCtx, cancel := context.WithTimeout(ctx, d.cfg.DownloadTimeout)
       defer cancel()

       req, err := http.NewRequestWithContext(downloadCtx, http.MethodGet, demoURL, nil)
       if err != nil { return fmt.Errorf("create request: %w", err) }
       req.Header.Set("User-Agent", "DemDownloader/1.0")

       // Use a fresh http.Client with the configured timeout
       client := &http.Client{Timeout: d.cfg.DownloadTimeout}
       resp, err := client.Do(req)
       if err != nil { return fmt.Errorf("http get: %w", err) }
       defer resp.Body.Close()

       if resp.StatusCode != http.StatusOK {
           return fmt.Errorf("http status %d", resp.StatusCode)
       }

       // Safety net: limit response body size (Pitfall 4)
       limitedBody := io.LimitReader(resp.Body, d.cfg.MaxBytes)

       // Stream directly from HTTP response to MinIO — NO io.ReadAll, NO temp files (D-04)
       _, err = d.minio.PutObject(downloadCtx, bucket, objectKey, limitedBody, -1,
           minio.PutObjectOptions{
               ContentType: "application/gzip",
               PartSize:    128 * 1024 * 1024, // 128 MiB parts for large files
           })
       if err != nil { return fmt.Errorf("minio put: %w", err) }

       return nil
   }
   ```

```go
package main

import (
    "context"
    "log/slog"
    "os"

    "github.com/alekseitsvetkov/dem/internal/downloader"
    "github.com/alekseitsvetkov/dem/internal/service"
    "github.com/alekseitsvetkov/dem/pkg/minio"
    "github.com/alekseitsvetkov/dem/pkg/natsutil"
)

func main() {
    cfg := downloader.LoadConfig()

    logger := slog.New(slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
        Level: parseLogLevel(cfg.LogLevel),
    }))

    nc, js, err := natsutil.NewNATSConn(
        natsutil.WithURL(cfg.NATSURL),
        natsutil.WithName("downloader"),
    )
    if err != nil {
        logger.Error("nats connect failed", slog.String("error", err.Error()))
        os.Exit(1)
    }
    defer nc.Close()

    minioClient, err := minio.NewMinioClient(
        minio.WithEndpoint(cfg.MinioEndpoint),
        minio.WithCredentials(cfg.MinioAccessKey, cfg.MinioSecretKey),
        // WithSSL conditionally: only if cfg.MinioUseSSL
    )
    if err != nil {
        logger.Error("minio connect failed", slog.String("error", err.Error()))
        os.Exit(1)
    }

    svc := downloader.NewDownloaderService(cfg,
        downloader.WithNATS(js),
        downloader.WithMinio(minioClient),
        downloader.WithLogger(logger),
    )

    runner := service.NewRunner(service.WithLogger(logger))
    runner.AddService(svc)

    if err := runner.Run(context.Background()); err != nil {
        logger.Error("downloader service exited with error", slog.String("error", err.Error()))
        os.Exit(1)
    }
}

func parseLogLevel(level string) slog.Level {
    switch level {
    case "debug": return slog.LevelDebug
    case "warn":  return slog.LevelWarn
    case "error": return slog.LevelError
    default:      return slog.LevelInfo
    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HLTV CDN -> Downloader | Untrusted .dem.gz binary crosses into the process via HTTP GET |
| Downloader -> MinIO | Demo file data crosses into object storage |
| Downloader -> NATS | Parse job payloads cross into the messaging system |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-08 | Denial of Service | HLTV CDN download | mitigate | `io.LimitReader` with MaxBytes=500MB prevents oversized response from exhausting memory (Pitfall 4). `context.WithTimeout` with DownloadTimeout (30min) prevents hung connections. Streaming upload means memory is proportional to buffer, not file size. |
| T-06-09 | Tampering | MinIO object key | mitigate | Object keys use match_id from the NATS message (validated by poller). No user-controlled path components. Key format `demos/{match_id}.dem.gz` is deterministic. |
| T-06-10 | Information Disclosure | MinIO credentials | mitigate | Credentials from environment variables via Viper. No hardcoded secrets. Default dev credentials (minioadmin/minioadmin) documented for local development only. |
| T-06-11 | Repudiation | Download completion | accept | The downloader is a pass-through — the NATS message Ack + MinIO object existence together provide an audit trail. No separate audit log needed at this stage. |
</threat_model>
